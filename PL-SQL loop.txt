1. Finding the biggest number by directly assigning values to variables if then loop.

	DECLARE
    x number(2);
    y number(2);
BEGIN
    x:=5;
    y:=10;
    IF x>y then
        dbms_output.put_line(x);
    ELSE 
        dbms_output.put_line(y);
    END IF;
END;

Output:
10

4. Program to find largest value array.
	DECLARE 
   type grades IS VARRAY(5) OF INTEGER; 
   marks grades; 
   total integer; 
   max_n integer;
BEGIN 
   marks:= grades(98, 97, 78, 87, 92); 
   total := marks.count; 
   max_n:=marks(1);
   FOR i in 1 .. total LOOP 
    if max_n < marks(i) then
        max_n:=marks(i);
    end if;
   END LOOP; 
   dbms_output.put_line(max_n);
END;

Output:
98

5. Program to show the effect of user defined exception
	DECLARE 
   c_id customers.id%type := &cc_id; 
   c_name customerS.Name%type; 
   c_addr customers.address%type;  
   -- user defined exception 
   ex_invalid_id  EXCEPTION; 
BEGIN 
   IF c_id <= 0 THEN 
      RAISE ex_invalid_id; 
   ELSE 
      SELECT  name, address INTO  c_name, c_addr 
      FROM customers 
      WHERE id = c_id;
      DBMS_OUTPUT.PUT_LINE ('Name: '||  c_name);  
      DBMS_OUTPUT.PUT_LINE ('Address: ' || c_addr); 
   END IF; 

EXCEPTION 
   WHEN ex_invalid_id THEN 
      dbms_output.put_line('ID must be greater than zero!'); 
   WHEN no_data_found THEN 
      dbms_output.put_line('No such customer!'); 
   WHEN others THEN 
      dbms_output.put_line('Error!');  
END;

Output:
Enter value for cc_id: -6 (let's enter a value -6)

new  2: c_id customers.id%type := -6; 
ID must be greater than zero!
